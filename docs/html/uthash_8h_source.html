00182 /* delete "delptr" from the hash table.
00183  * "the usual" patch-up process for the app-order doubly-linked-list.
00184  * The use of _hd_hh_del below deserves special explanation.
00185  * These used to be expressed using (delptr) but that led to a bug
00186  * if someone used the same symbol for the head and deletee, like
00187  *  HASH_DELETE(hh,users,users);
00188  * We want that to work, but by changing the head (users) below
00189  * we were forfeiting our ability to further refer to the deletee (users)
00190  * in the patch-up process. Solution: use scratch space to
00191  * copy the deletee pointer, then the latter references are via that
00192  * scratch pointer rather than through the repointed (users) symbol.
00193  */
00194 #define HASH_DELETE(hh,head,delptr)                                              \
00195 do {                                                                             \
00196     unsigned _hd_bkt;                                                            \
00197     struct UT_hash_handle *_hd_hh_del;                                           \
00198     if ( ((delptr)->hh.prev == NULL) && ((delptr)->hh.next == NULL) )  {         \
00199         uthash_free((head)->hh.tbl->buckets,                                     \
00200                     (head)->hh.tbl->num_buckets*sizeof(struct UT_hash_bucket) ); \
00201         HASH_BLOOM_FREE((head)->hh.tbl);                                         \
00202         uthash_free((head)->hh.tbl, sizeof(UT_hash_table));                      \
00203         head = NULL;                                                             \
00204     } else {                                                                     \
00205         _hd_hh_del = &((delptr)->hh);                                            \
00206         if ((delptr) == ELMT_FROM_HH((head)->hh.tbl,(head)->hh.tbl->tail)) {     \
00207             (head)->hh.tbl->tail =                                               \
00208                 (UT_hash_handle*)((char*)((delptr)->hh.prev) +                   \
00209                 (head)->hh.tbl->hho);                                            \
00210         }                                                                        \
00211         if ((delptr)->hh.prev) {                                                 \
00212             ((UT_hash_handle*)((char*)((delptr)->hh.prev) +                      \
00213                     (head)->hh.tbl->hho))->next = (delptr)->hh.next;             \
00214         } else {                                                                 \
00215             DECLTYPE_ASSIGN(head,(delptr)->hh.next);                             \
00216         }                                                                        \
00217         if (_hd_hh_del->next) {                                                  \
00218             ((UT_hash_handle*)((char*)_hd_hh_del->next +                         \
00219                     (head)->hh.tbl->hho))->prev =                                \
00220                     _hd_hh_del->prev;                                            \
00221         }                                                                        \
00222         HASH_TO_BKT( _hd_hh_del->hashv, (head)->hh.tbl->num_buckets, _hd_bkt);   \
00223         HASH_DEL_IN_BKT(hh,(head)->hh.tbl->buckets[_hd_bkt], _hd_hh_del);        \
00224         (head)->hh.tbl->num_items--;                                             \
00225     }                                                                            \
00226     HASH_FSCK(hh,head);                                                          \
00227 } while (0)

00251 #define HASH_FSCK(hh,head)                                                       \
00252 do {                                                                             \
00253     unsigned _bkt_i;                                                             \
00254     unsigned _count, _bkt_count;                                                 \
00255     char *_prev;                                                                 \
00256     struct UT_hash_handle *_thh;                                                 \
00257     if (head) {                                                                  \
00258         _count = 0;                                                              \
00259         for( _bkt_i = 0; _bkt_i < (head)->hh.tbl->num_buckets; _bkt_i++) {       \
00260             _bkt_count = 0;                                                      \
00261             _thh = (head)->hh.tbl->buckets[_bkt_i].hh_head;                      \
00262             _prev = NULL;                                                        \
00263             while (_thh) {                                                       \
00264                if (_prev != (char*)(_thh->hh_prev)) {                            \
00265                    HASH_OOPS("invalid hh_prev %p, actual %p\n",                  \
00266                     _thh->hh_prev, _prev );                                      \
00267                }                                                                 \
00268                _bkt_count++;                                                     \
00269                _prev = (char*)(_thh);                                            \
00270                _thh = _thh->hh_next;                                             \
00271             }                                                                    \
00272             _count += _bkt_count;                                                \
00273             if ((head)->hh.tbl->buckets[_bkt_i].count !=  _bkt_count) {          \
00274                HASH_OOPS("invalid bucket count %d, actual %d\n",                 \
00275                 (head)->hh.tbl->buckets[_bkt_i].count, _bkt_count);              \
00276             }                                                                    \
00277         }                                                                        \
00278         if (_count != (head)->hh.tbl->num_items) {                               \

00280                 (head)->hh.tbl->num_items, _count );                             \
00281         }                                                                        \
00282         /* traverse hh in app order; check next/prev integrity, count */         \
00283         _count = 0;                                                              \
00284         _prev = NULL;                                                            \
00285         _thh =  &(head)->hh;                                                     \
00286         while (_thh) {                                                           \

00294                                   (head)->hh.tbl->hho) : NULL );                 \

00296         if (_count != (head)->hh.tbl->num_items) {                               \
00297             HASH_OOPS("invalid app item count %d, actual %d\n",                  \
00298                 (head)->hh.tbl->num_items, _count );                             \

00300     }                                                                            \
00301 } while (0)

00392   char *_hj_key=(char*)(key);                                                    \

00395   _hj_k = keylen;                                                                \
00396   while (_hj_k >= 12) {                                                          \
00397     _hj_i +=    (_hj_key[0] + ( (unsigned)_hj_key[1] << 8 )                      \
00398         + ( (unsigned)_hj_key[2] << 16 )                                         \
00399         + ( (unsigned)_hj_key[3] << 24 ) );                                      \

00401         + ( (unsigned)_hj_key[6] << 16 )                                         \
00402         + ( (unsigned)_hj_key[7] << 24 ) );                                      \
00403     hashv += (_hj_key[8] + ( (unsigned)_hj_key[9] << 8 )                         \
00404         + ( (unsigned)_hj_key[10] << 16 )                                        \
00405         + ( (unsigned)_hj_key[11] << 24 ) );                                     \
00406                                                                                  \

00408                                                                                  \
00409      _hj_key += 12;                                                              \
00410      _hj_k -= 12;                                                                \
00411   }                                                                              \
00412   hashv += keylen;                                                               \
00413   switch ( _hj_k ) {                                                             \
00414      case 11: hashv += ( (unsigned)_hj_key[10] << 24 );                          \
00415      case 10: hashv += ( (unsigned)_hj_key[9] << 16 );                           \
00416      case 9:  hashv += ( (unsigned)_hj_key[8] << 8 );                            \
00417      case 8:  _hj_j += ( (unsigned)_hj_key[7] << 24 );                           \
00418      case 7:  _hj_j += ( (unsigned)_hj_key[6] << 16 );                           \
00419      case 6:  _hj_j += ( (unsigned)_hj_key[5] << 8 );                            \
00420      case 5:  _hj_j += _hj_key[4];                                               \
00421      case 4:  _hj_i += ( (unsigned)_hj_key[3] << 24 );                           \
00422      case 3:  _hj_i += ( (unsigned)_hj_key[2] << 16 );                           \
00423      case 2:  _hj_i += ( (unsigned)_hj_key[1] << 8 );                            \

00608 /* Bucket expansion has the effect of doubling the number of buckets
00609  * and redistributing the items into the new buckets. Ideally the
00610  * items will distribute more or less evenly into the new buckets
00611  * (the extent to which this is true is a measure of the quality of
00612  * the hash function as it applies to the key domain). 
00613  * 
00614  * With the items distributed into more buckets, the chain length
00615  * (item count) in each bucket is reduced. Thus by expanding buckets
00616  * the hash keeps a bound on the chain length. This bounded chain 
00617  * length is the essence of how a hash provides constant time lookup.
00618  * 
00619  * The calculation of tbl->ideal_chain_maxlen below deserves some
00620  * explanation. First, keep in mind that we're calculating the ideal
00621  * maximum chain length based on the *new* (doubled) bucket count.
00622  * In fractions this is just n/b (n=number of items,b=new num buckets).
00623  * Since the ideal chain length is an integer, we want to calculate 
00624  * ceil(n/b). We don't depend on floating point arithmetic in this
00625  * hash, so to calculate ceil(n/b) with integers we could write
00626  * 
00627  *      ceil(n/b) = (n/b) + ((n%b)?1:0)
00628  * 
00629  * and in fact a previous version of this hash did just that.
00630  * But now we have improved things a bit by recognizing that b is
00631  * always a power of two. We keep its base 2 log handy (call it lb),
00632  * so now we can write this with a bit shift and logical AND:
00633  * 
00634  *      ceil(n/b) = (n>>lb) + ( (n & (b-1)) ? 1:0)
00635  * 
00636  */
00637 #define HASH_EXPAND_BUCKETS(tbl)                                                 \
00638 do {                                                                             \
00639     unsigned _he_bkt;                                                            \
00640     unsigned _he_bkt_i;                                                          \
00641     struct UT_hash_handle *_he_thh, *_he_hh_nxt;                                 \
00642     UT_hash_bucket *_he_new_buckets, *_he_newbkt;                                \
00643     _he_new_buckets = (UT_hash_bucket*)uthash_malloc(                            \
00644              2 * tbl->num_buckets * sizeof(struct UT_hash_bucket));              \
00645     if (!_he_new_buckets) { uthash_fatal( "out of memory"); }                    \
00646     memset(_he_new_buckets, 0,                                                   \
00647             2 * tbl->num_buckets * sizeof(struct UT_hash_bucket));               \
00648     tbl->ideal_chain_maxlen =                                                    \
00649        (tbl->num_items >> (tbl->log2_num_buckets+1)) +                           \
00650        ((tbl->num_items & ((tbl->num_buckets*2)-1)) ? 1 : 0);                    \
00651     tbl->nonideal_items = 0;                                                     \
00652     for(_he_bkt_i = 0; _he_bkt_i < tbl->num_buckets; _he_bkt_i++)                \
00653     {                                                                            \
00654         _he_thh = tbl->buckets[ _he_bkt_i ].hh_head;                             \
00655         while (_he_thh) {                                                        \
00656            _he_hh_nxt = _he_thh->hh_next;                                        \
00657            HASH_TO_BKT( _he_thh->hashv, tbl->num_buckets*2, _he_bkt);            \
00658            _he_newbkt = &(_he_new_buckets[ _he_bkt ]);                           \
00659            if (++(_he_newbkt->count) > tbl->ideal_chain_maxlen) {                \
00660              tbl->nonideal_items++;                                              \
00661              _he_newbkt->expand_mult = _he_newbkt->count /                       \
00662                                         tbl->ideal_chain_maxlen;                 \
00663            }                                                                     \
00664            _he_thh->hh_prev = NULL;                                              \
00665            _he_thh->hh_next = _he_newbkt->hh_head;                               \
00666            if (_he_newbkt->hh_head) _he_newbkt->hh_head->hh_prev =               \
00667                 _he_thh;                                                         \
00668            _he_newbkt->hh_head = _he_thh;                                        \
00669            _he_thh = _he_hh_nxt;                                                 \
00670         }                                                                        \
00671     }                                                                            \
00672     uthash_free( tbl->buckets, tbl->num_buckets*sizeof(struct UT_hash_bucket) ); \
00673     tbl->num_buckets *= 2;                                                       \
00674     tbl->log2_num_buckets++;                                                     \
00675     tbl->buckets = _he_new_buckets;                                              \
00676     tbl->ineff_expands = (tbl->nonideal_items > (tbl->num_items >> 1)) ?         \
00677         (tbl->ineff_expands+1) : 0;                                              \

00679         tbl->noexpand=1;                                                         \
